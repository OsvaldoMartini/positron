    if (isSaving) {
      return "green100"

      return "black100"
    } else {
      return "red100"

  getSaveText = () => {
    const { article, edit } = this.props
    const { isSaving } = edit
 
    if (isSaving) {

      return "Save Article"

    if (isPublishing && isPublished) {

      return "Publishing..."
    } else if (isPublished) {
      return "Unpublish"

      return "Publish"
    }
  }
